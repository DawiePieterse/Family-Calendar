        document.getElementById('activityForm').addEventListener('submit', async function(e) {

            e.preventDefault();

            const t = translations[currentLanguage];


            if (!document.getElementById('childName').value) {

                showError(t.errorSelectChild);

                return;

            }


            const dropoffChecked = document.querySelectorAll('input[name="dropoffResponsible"]:checked');

            const pickupChecked = document.querySelectorAll('input[name="pickupResponsible"]:checked');

            const dropoffOther = document.getElementById('dropoffOther').value.trim();

            const pickupOther = document.getElementById('pickupOther').value.trim();


            if (dropoffChecked.length === 0 && !dropoffOther) {

                showError(t.errorSelectDropoff);

                return;

            }

            if (pickupChecked.length === 0 && !pickupOther) {

                showError(t.errorSelectPickup);

                return;

            }


            const startTime = document.getElementById('startTime').value;

            const endTime = document.getElementById('endTime').value;

            if (endTime <= startTime) {

                showError(t.errorEndTime);

                return;

            }


            if (document.getElementById('isRecurring').checked) {

                const endDate = document.getElementById('recurrenceEndDate').value;

                if (!endDate) {

                    showError(t.errorEndDate);

                    return;

                }

                const startDate = document.getElementById('activityDate').value;

                if (endDate < startDate) {

                    showError(t.errorDateOrder);

                    return;

                }

                const daysChecked = document.querySelectorAll('input[name="recurDays"]:checked');

                    showError(t.errorRecurDays);

            }

            if (dropoffOther) dropoffResponsible.push(dropoffOther);
